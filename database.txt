CREATE TABLE conversationdetails(
    user_input VARCHAR(255) NOT NULL,
    response VARCHAR(255) NOT NULL
);

CREATE TABLE trainstation(
    station_name VARCHAR(50) PRIMARY KEY UNIQUE NOT NULL,
    abbreviations VARCHAR(50) UNIQUE NOT NULL
)

CREATE TABLE trainperformance(
  rid char(16), 
  tpl char(7), 
  pta char(5), 
  ptd char(5), 
  wta char(8), 
  wtp char(8), 
  wtd char(8), 
  arr_et char(5), 
  arr_wet char(5), 
  arr_atremoved boolean, 
  pass_et char(5), 
  pass_wet char(5), 
  pass_atremoved boolean, 
  dep_et char(5), 
  dep_wet char(5), 
  dep_atremoved boolean, 
  arr_at char(5), 
  pass_at char(5), 
  dep_at char(5), 
  cr_code smallint, 
  lr_code smallint)
    
CREATE TABLE weather(
  name char(15), 
  datetime date, 
  tempmax decimal, 
  tempmin decimal, 
  temp decimal, 
  feelslikemax decimal, 
  feelslikemin decimal, 
  feelslike decimal, 
  dew decimal, 
  humidity decimal, 
  precip decimal, 
  precipprob char(5), 
  precipcover decimal, 
  preciptype char(15), 
  snow char(15), 
  snowdepth decimal, 
  windgust decimal, 
  windspeed decimal, 
  winddir decimal, 
  sealevelpressure decimal,
  cloudcover decimal, 
  visibility decimal, 
  solarradiation char(5),
  solarenergy char(5), 
  uvindex smallint, 
  severerisk char(5),
  sunrise char(25),
  sunset char(25), 
  moonphase decimal, 
  conditions char(50),
  description char(255),
  icon char(50), 
  stations char(255));

--To add the data into the database table, right click on the table created in the schema and choose --import\export. In Options, change export to import. In file info section, choose the file you need --and change the file format to csv. In miscellaneous section, turn the header to yes and choose ',' --as the delimiter
